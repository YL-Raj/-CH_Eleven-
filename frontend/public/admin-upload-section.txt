<script>
let selectedFile = null;
let parsedData = null;

function handleDrop(event) {
  event.preventDefault();
  event.stopPropagation();
  event.target.style.background = '#f0f8ff';
  
  const files = event.dataTransfer.files;
  if (files.length > 0) {
    selectedFile = files[0];
    handleFilePreview();
  }
}

function handleFileSelect(event) {
  selectedFile = event.target.files[0];
  if (selectedFile) {
    handleFilePreview();
  }
}

function handleFilePreview() {
  if (!selectedFile) return;
  
  const reader = new FileReader();
  reader.onload = (e) => {
    try {
      const content = e.target.result;
      const ext = selectedFile.name.split('.').pop().toLowerCase();
      
      if (ext === 'json') {
        parsedData = JSON.parse(content);
      } else if (ext === 'csv') {
        // Send to backend for parsing
        parseCSV(content);
        return;
      }
      
      showFilePreview();
    } catch (err) {
      showValidationError(`Invalid ${ext.toUpperCase()}: ${err.message}`);
    }
  };
  reader.readAsText(selectedFile);
}

function parseCSV(content) {
  // This will be processed server-side
  // For now, show that CSV is ready
  parsedData = { csv_content: content };
  showFilePreview();
}

function showFilePreview() {
  if (!parsedData) return;
  
  document.getElementById('file-preview').style.display = 'block';
  
  // Extract preview data
  const summary = parsedData.match_summary || {};
  const squads = parsedData.squads || {};
  const innings = parsedData.innings || {};
  
  const team1Players = squads.team1?.players?.length || 0;
  const team2Players = squads.team2?.players?.length || 0;
  const inningsRecs = (innings.first_innings?.batting?.length || 0) + (innings.second_innings?.batting?.length || 0);
  
  document.getElementById('preview-match').textContent = summary.match || 'N/A';
  document.getElementById('preview-teams').textContent = `${summary.team1_name || '?'} vs ${summary.team2_name || '?'}`;
  document.getElementById('preview-players').textContent = `${team1Players + team2Players} players`;
  document.getElementById('preview-innings').textContent = `${inningsRecs} batting records`;
  
  // Show validation
  const validation = parsedData.validation || {};
  const msgDiv = document.getElementById('validation-messages');
  let html = '';
  
  if (validation.errors && validation.errors.length > 0) {
    html += `<div class="error-msg">⚠ Errors: ${validation.errors.join(', ')}</div>`;
  }
  if (validation.warnings && validation.warnings.length > 0) {
    html += `<div style="color: #ff9800; font-size: 12px; margin-top: 5px;">⚡ Warnings: ${validation.warnings.join(', ')}</div>`;
  }
  if (validation.valid) {
    html += `<div class="success-msg">✓ All validations passed</div>`;
  }
  
  msgDiv.innerHTML = html;
}

function clearFilePreview() {
  document.getElementById('file-preview').style.display = 'none';
  document.getElementById('upload-status').innerHTML = '';
  selectedFile = null;
  parsedData = null;
  document.getElementById('file-input').value = '';
}

async function uploadMatchData() {
  if (!selectedFile || !parsedData) {
    alert('No file selected');
    return;
  }
  
  const ext = selectedFile.name.split('.').pop().toLowerCase();
  const isCSV = ext === 'csv';
  
  // Show progress
  document.getElementById('upload-progress').style.display = 'block';
  document.getElementById('upload-btn').disabled = true;
  
  try {
    const formData = new FormData();
    formData.append('file', selectedFile);
    formData.append('format', isCSV ? 'csv' : 'json');
    formData.append('data', isCSV ? null : JSON.stringify(parsedData));
    
    // Simulate upload progress
    let progress = 0;
    const progressInterval = setInterval(() => {
      if (progress < 90) {
        progress += Math.random() * 30;
        updateProgress(Math.min(progress, 90));
      }
    }, 200);
    
    const res = await fetch(`${API}/match/bulk-upload`, {
      method: 'POST',
      headers: { 'x-admin-key': state.adminKey },
      body: formData,
    });
    
    clearInterval(progressInterval);
    updateProgress(100);
    
    const result = await res.json();
    
    if (!res.ok) {
      showUploadError(result.error || 'Upload failed');
      return;
    }
    
    // Success
    setTimeout(() => {
      showUploadSuccess(result);
      clearFilePreview();
      fetchAll();
    }, 500);
    
  } catch (e) {
    showUploadError(e.message);
  } finally {
    document.getElementById('upload-btn').disabled = false;
    document.getElementById('upload-progress').style.display = 'none';
  }
}

function updateProgress(pct) {
  document.getElementById('progress-bar').style.width = pct + '%';
  document.getElementById('progress-text').textContent = Math.round(pct) + '%';
}

function showValidationError(msg) {
  const div = document.getElementById('validation-messages');
  div.innerHTML = `<div class="error-msg">${msg}</div>`;
  document.getElementById('file-preview').style.display = 'block';
}

function showUploadSuccess(result) {
  const div = document.getElementById('upload-status');
  div.innerHTML = `
    <div class="success-msg">
      ✓ Upload successful!
      ${result.players_created ? `<br>Players created: ${result.players_created}` : ''}
      ${result.stats_updated ? `<br>Stats updated: ${result.stats_updated}` : ''}
    </div>
  `;
}

function showUploadError(msg) {
  const div = document.getElementById('upload-status');
  div.innerHTML = `<div class="error-msg">⚠ ${msg}</div>`;
}

function startNewGame() {
  if (!confirm('Start new match? This will reset current match data.')) return;
  
  const matchName = prompt('Enter match name:');
  if (!matchName) return;
  
  const team1 = prompt('Enter Team 1 name:');
  if (!team1) return;
  
  const team2 = prompt('Enter Team 2 name:');
  if (!team2) return;
  
  // Save match info
  document.getElementById('m-name').value = matchName;
  document.getElementById('m-ta').value = team1;
  document.getElementById('m-tb').value = team2;
  document.getElementById('m-status').value = 'upcoming';
  
  saveMatch();
  
  document.getElementById('match-status-display').textContent = `${matchName}: ${team1} vs ${team2} (Upcoming)`;
}
</script>
